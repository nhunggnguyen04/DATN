\chapter{Phân tích và thiết kế hệ thống}
\label{ch:analysis-design}

\section{Đề xuất kiến trúc giải pháp}
\label{sec:solution-arch}

Dựa trên những phân tích về hiện trạng và các vấn đề tồn đọng ở Chương 2, đồ án đề xuất giải pháp xây dựng hệ thống dữ liệu theo hướng \textbf{Modern Data Stack triển khai nội bộ (On-premise)} kết hợp với mô hình lưu trữ đa tầng và chiến lược nạp \textbf{ELT}. Đề xuất này dựa trên các cơ sở thiết kế sau:

\subsection{Lựa chọn kiến trúc tổng thể và chiến lược nạp}

\begin{itemize}
    \item \textbf{Kiến trúc Modern Data Stack On-premise:}
    \begin{itemize}
        \item \textit{Đáp ứng toàn diện yêu cầu:} Khắc phục hoàn toàn những giới hạn của kiến trúc Legacy và Hybrid, hệ thống mới đáp ứng đầy đủ các yêu cầu chức năng phức tạp như lưu vết lịch sử, tự động hóa quy trình, kiểm soát chất lượng dữ liệu và xử lý dữ liệu phi cấu trúc.
        \item \textit{Tối ưu chi phí:} Toàn bộ ngăn xếp (stack) sử dụng các công nghệ mã nguồn mở (Open-source), giúp hệ thống không phát sinh các khoản chi phí bản quyền đắt đỏ.
        \item \textit{Tuân thủ pháp lý:} Việc vận hành hoàn toàn trên hạ tầng nội bộ (on-premise) đảm bảo tuân thủ tuyệt đối quy định về lưu trữ dữ liệu cá nhân trong nước theo Nghị định 13/2023/NĐ-CP.
    \end{itemize}

    \item \textbf{Chiến lược nạp ELT (Extract $\rightarrow$ Load $\rightarrow$ Transform):}
    \begin{itemize}
        \item Hệ thống thực hiện trích xuất và nạp (Load) toàn bộ dữ liệu thô vào tầng Bronze trước, sau đó mới tiến hành biến đổi (Transform) bằng công cụ dbt ngay bên trong kho dữ liệu.
        \item Chiến lược này giúp phân tách rạch ròi trách nhiệm của các công cụ: Python chỉ chuyên trách Extract \& Load, trong khi dbt chỉ tập trung xử lý Transform. Điều này giúp hệ thống cực kỳ dễ bảo trì và gỡ lỗi (debug).
    \end{itemize}
\end{itemize}

\subsection{Lựa chọn mô hình tổ chức dữ liệu}

Hệ thống ứng dụng \textbf{kiến trúc Medallion (Bronze $\rightarrow$ Silver $\rightarrow$ Gold)} để phân tầng dữ liệu theo mức độ tinh sạch, mỗi tầng phục vụ một mục đích và nhóm người dùng riêng biệt. Trong đó, tầng \textbf{Bronze} đóng vai trò giữ lại dữ liệu thô nguyên bản, cho phép hệ thống dễ dàng tái xử lý lại thông tin khi logic nghiệp vụ thay đổi mà không cần phải trích xuất lại từ đầu. Các tầng tiếp theo được mô hình hóa chi tiết như sau:

\begin{itemize}
    \item \textbf{Mô hình hóa tầng Silver (Data Vault 2.0):}
    \begin{itemize}
        \item \textit{Lưu trữ lịch sử toàn diện:} Sử dụng bảng Satellite kết hợp kỹ thuật SCD Type 2 để ghi nhận mọi thay đổi, giúp hệ thống có khả năng trả lời chính xác câu hỏi "Dữ liệu tại thời điểm X trong quá khứ là gì?".
        \item \textit{Tính linh hoạt và mở rộng:} Cấu trúc tách biệt giữa khóa (Hub/Link) và thuộc tính (Satellite) giúp việc bổ sung thêm nguồn dữ liệu hoặc các trường thông tin mới không làm phá vỡ mô hình hiện hành.
        \item \textit{Tối ưu hiệu suất:} Thiết kế các bảng độc lập cho phép điều phối viên nạp dữ liệu song song thông qua cơ chế Pool, giúp giảm thiểu đáng kể tổng thời gian chạy luồng.
    \end{itemize}

    \item \textbf{Mô hình hóa tầng Gold (Star Schema):}
    \begin{itemize}
        \item \textit{Tối ưu truy vấn và báo cáo:} Với bảng sự kiện (Fact table) ở trung tâm và các bảng chiều (Dimension) bao quanh, Star Schema tạo ra cơ chế lọc chéo (cross-filter) tự nhiên trong Power BI. Thiết kế này hạn chế các phép kết nối (join) phức tạp, mang lại tốc độ truy vấn vượt trội.
        \item \textit{Phục vụ trực tiếp nghiệp vụ:} Cấu trúc mô hình sao được thiết kế để giải quyết trực tiếp chủ đề phân tích đã xác định ở Chương 2.
    \end{itemize}
\end{itemize}

\subsection{Pipeline dữ liệu phi cấu trúc (OCR)}
Do đặc thù ngành ngân hàng, các chứng từ như Căn cước công dân (CCCD) hay Sổ tiết kiệm chứa đựng lượng lớn dữ liệu cá nhân nhạy cảm. Việc sử dụng hình ảnh thực tế của khách hàng cho quá trình phát triển, thử nghiệm và trình diễn là vi phạm nghiêm trọng các quy định về bảo mật, đặc biệt là Nghị định 13/2023/NĐ-CP về bảo vệ dữ liệu cá nhân.

Để giải quyết triệt để rào cản pháp lý này, toàn bộ hình ảnh đầu vào được sử dụng để chạy thực nghiệm trong pipeline OCR đều là \textbf{dữ liệu giả lập}. Cụ thể, một tập lệnh Python tự động hóa đã được xây dựng để sinh ra các hình ảnh chứng từ dựa trên phôi mẫu chuẩn của nhà nước và ngân hàng. 

Quá trình giả lập này được thực hiện qua hai bước tinh vi:
\begin{itemize}
    \item \textit{Sinh nội dung ngẫu nhiên:} Điền các thông tin định danh hoàn toàn giả lập (họ tên, số ID, địa chỉ, số tiền...) vào các trường tương ứng trên phôi mẫu.
    \item \textit{Mô phỏng điều kiện thực tế:} Chủ động thêm vào các hiệu ứng nhiễu hình học và quang học bằng thuật toán (như làm mờ, xoay nghiêng, nhiễu hạt, chói sáng cục bộ hay đổ bóng).
\end{itemize}

Về chi tiết hoạt động, hai luồng xử lý này có những đặc tả nghiệp vụ khác nhau:

\begin{itemize}
    \item \textbf{Pipeline xử lý ảnh Căn cước công dân (CCCD):}
    \begin{itemize}
        \item \textit{Tiền xử lý ảnh:} Hình ảnh chụp CCCD thường bị nhiễu do ánh sáng chói hoặc bóng râm. Hệ thống áp dụng thuật toán CLAHE để cân bằng độ tương phản, kết hợp Non-local Means để khử nhiễu hạt. Tiếp theo, thuật toán Canny Edge Detection và Perspective Transform được dùng để nắn phẳng tấm thẻ từ góc chụp nghiêng về dạng hình chữ nhật vuông vức.
        \item \textit{Trích xuất thông tin:} Mô hình nhận dạng quang học quét toàn bộ ảnh, định vị văn bản và kết hợp với các biểu thức chính quy (Regex) để trích xuất 9 trường dữ liệu cốt lõi (họ tên, ngày sinh, số CCCD, v.v.). Các kết quả có điểm tin cậy (Confidence Score) đạt yêu cầu sẽ được nạp trực tiếp vào bảng \texttt{id\_card\_results} tại tầng Bronze.
    \end{itemize}

    \item \textbf{Pipeline xử lý ảnh Sổ tiết kiệm:}
    \begin{itemize}
        \item \textit{Tiền xử lý ảnh:} Tương tự như CCCD, hệ thống làm sạch và chuẩn hóa hình ảnh sổ tiết kiệm. Tuy nhiên, luồng này bỏ qua các bước không cần thiết đối với giấy tờ dạng khổ lớn như chuẩn hóa chiều hay ngưỡng hóa thích nghi phức tạp, giúp giảm bớt tài nguyên và tăng tốc độ tính toán.
        \item \textit{Trích xuất thông tin:} Do định dạng sổ tiết kiệm đa dạng hơn, hệ thống ưu tiên bóc tách dựa vào 7 vùng định sẵn tương ứng với các trường thông tin (như số tài khoản, số tiền gửi, lãi suất, kỳ hạn). Hệ thống cũng tự động thử nghiệm nhiều biến thể trích xuất và giữ lại kết quả có điểm tin cậy cao nhất trước khi nạp vào bảng \texttt{savings\_book\_results}.
    \end{itemize}
\end{itemize}

\subsection{Schema Audit}

Bên cạnh các schema phục vụ lưu trữ dữ liệu nghiệp vụ, hệ thống còn thiết kế riêng một schema \texttt{audit} để phục vụ công tác quản trị và vận hành. 

\textbf{Mục đích của Schema Audit:}
\begin{itemize}
    \item Ghi lại toàn bộ lịch sử thực thi của mọi task trong pipeline (bao gồm thông tin luồng chạy, thời gian thực thi và kết quả đầu ra).
    \item Hỗ trợ đắc lực cho kỹ sư dữ liệu trong việc theo dõi, gỡ lỗi và đánh giá độ tin cậy cũng như hiệu năng hoạt động của hệ thống.
\end{itemize}

\textbf{Cấu trúc bảng chính: \texttt{audit.pipeline\_run\_\allowbreak log}}

Đây là bảng nhật ký trung tâm, được thiết kế để lưu vết mọi hoạt động của luồng dữ liệu. Cấu trúc bảng được chia thành các nhóm cột chi tiết như sau:

\begin{itemize}
    \item \textbf{Nhóm định danh task:} Gồm các cột \texttt{dag\_id}, \texttt{task\_id}, \texttt{run\_date}, \texttt{attempt}. Nhóm này giúp định danh chính xác tác vụ đang chạy, thuộc luồng nào và ở lần chạy thứ mấy.
    \item \textbf{Nhóm trạng thái:} Cột \texttt{status} ghi nhận tình trạng hiện tại của task với các giá trị chuẩn như: \texttt{started}, \texttt{success}, \texttt{failed}, \texttt{skipped}.
    \item \textbf{Nhóm thời gian:} Gồm \texttt{started\_at}, \texttt{ended\_at} và \texttt{duration\_sec}. Trong đó, \texttt{duration\_sec} là cột tính toán tự động giúp đo lường thời gian chạy thực tế của task.
    \item \textbf{Nhóm thống kê tác động dữ liệu:} Gồm \texttt{rows\_processed}, \texttt{rows\_inserted}, \texttt{rows\_updated}, \texttt{rows\_deleted}. Nhóm cột này cung cấp số liệu thực tế về khối lượng bản ghi được xử lý qua từng bước.
    \item \textbf{Nhóm thông tin lỗi \& debug:} Bao gồm \texttt{error\_message}, \texttt{log\_file\_path}, \texttt{airflow\_log\_url}, \texttt{host\_name} và đặc biệt là \texttt{extra\_metadata} (lưu dưới dạng định dạng JSON tự do cho phép mở rộng linh hoạt các thông tin log đặc thù mà không cần sửa bảng).
\end{itemize}

Để tối ưu hóa việc truy xuất và trực quan hóa tình trạng "sức khỏe" của toàn bộ hệ thống pipeline, schema \texttt{audit} được thiết kế thêm các View tổng hợp dữ liệu từ bảng log chính.

\textbf{Các View hỗ trợ phân tích:}
\begin{itemize}
    \item \texttt{audit.v\_latest\_run\_\allowbreak per\_task}: Cung cấp trạng thái thực thi mới nhất của từng task. View này đóng vai trò như một bảng điều khiển thời gian thực, giúp kỹ sư dữ liệu nắm bắt nhanh chóng hiện trạng của hệ thống (hiện có task nào đang treo hay vừa thất bại không).
    \item \texttt{audit.v\_task\_success\_\allowbreak rate\_30d}: Chuyên dùng để tính toán tỷ lệ thành công  và thống kê thời gian chạy (trung bình, tối đa) của các task trong vòng 30 ngày gần nhất. Tập dữ liệu từ view này đặc biệt quan trọng trong việc phát hiện các nút thắt cổ chai về hiệu năng và đánh giá độ ổn định dài hạn của các luồng xử lý.
\end{itemize}

\subsection{Cơ chế cảnh báo tự động}

Để đảm bảo tính chủ động trong công tác vận hành và rút ngắn thời gian xử lý sự cố, hệ thống điều phối Apache Airflow được tích hợp cơ chế thông báo tự động thông qua các hàm callback. Cơ chế này kết nối trực tiếp với các kênh giao tiếp của Email và hoạt động dựa trên 3 kịch bản chính:

\begin{itemize}
    \item \textbf{Cảnh báo thất bại} Bất kỳ tác vụ nào gặp lỗi và thất bại trong quá trình thực thi sẽ ngay lập tức kích hoạt callback để bắn cảnh báo khẩn cấp qua \textbf{Email}. Thông báo này thường đính kèm các ngữ cảnh quan trọng như tên luồng, tên tác vụ bị lỗi và đường dẫn (URL) trực tiếp tới file log để kỹ sư có thể truy cập và gỡ lỗi ngay lập tức.
    
    \item \textbf{Thông báo thử lại} Đối với các lỗi có thể do nguyên nhân khách quan tạm thời (như gián đoạn mạng, timeout), Airflow được cấu hình để thử lại. Khi quá trình này diễn ra, hệ thống sẽ báo cho đội ngũ biết rằng một tác vụ đang tự động khắc phục lỗi, tránh việc báo động giả làm trôi tin nhắn.
    
    \item \textbf{Xác nhận thành công:} Khi điểm neo cuối cùng của DAG hoàn thành, tức là toàn bộ pipeline đã chạy trọn vẹn và dữ liệu đã an toàn cập bến tầng Gold, một thông báo sẽ được gửi đi. Đây là tín hiệu xác nhận kho dữ liệu đã sẵn sàng để phục vụ cho các dashboard và báo cáo phân tích trong ngày.
\end{itemize}
